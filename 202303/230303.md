### Extensions 확장

형식 확장, 속성 메소드, 생성자 같은 멤버들 형식에 추가한다.

### 클래스 구조체 열거형 프로토콜

- Overridding 불가능

```swift
extension Type {
    copmutedProperty //계산 속성만 가능
    computedTypeProperty
    instanceMethod
    typeMethod
    initializer
    subscript
    NestedTyped
    //간편생상자만, 생성자 델리케이션이 정상적으로 실행되도록 해야한다. 
    //지정생성자와 소멸자는 원본 형식에서 구현 아니면 서브클래스 해야한다.
}

extension Type: Protocol, ... {
    requirements
}
```

```swift
struct Size {
    var width = 0.0
    var height = 0.0
}

extension Size {
    var area: Double {
        return width * height
    }    
}

let s = Size()
s.width
s.hegiht
s.area
```

```swift
extension Size: Equatable {
    public static func == (lhs: Size, rhs: Size) -> Bool {
    {
        return lhs.width == rhs.width && lhs.height == rhs.height
    }
}
```

### Adding Properties

- 계산 속성만 가능하다.
- 오버라이딩 불가능하다.
- 저장 속성이나 프로폴티 옵져버를 추가하는것을 불가능하다.

```swift
extension Date {
    var year: Int {
        let cal = Calendar.current
        return cal.component(.year, from: self)
    }
}

let today = Date()
today.year //2019
```
```swift

extension Double {
    var radianValue: Double {
            return (Double.pi * self) / 180.0
    }

    var degreeValue: Double {
        return self * 180.0 / Double.pi
    }
}

let dv = 45.0
dv.radianValue
dv.radianValue.degreeValue

```
### Adding Initializers

```swift
struct Size {
    var width = 0.0
    var height = 0.0

    init(value: Double) {
        width = value
        height = value
    }
} 

Size()//에러
Size(width: 12, height: 34) //에러
//파라미터가 없는 생성자, 파라미터가 두개인 생성자 생성해야한다. 하지만 extension사용 하면 된다

struct Size {
    var width = 0.0
    var height = 0.0
} 

extension Size {
    init(value: Double) {
        width = value
        height = value
    }
}

Size()
Size(width: 12, height: 34) 
Size(value: 12)
```

### Adding Methods

```swift
extension Double {
    func toFahrenhit() -> Double {
        return self * 9 / 5 +32
    }

    func toCelsius() -> Double {
        return (self - 32) * 5 / 9
    }

    static func converToFahrenheit(from celsius: Double) -> Double {
        return celsius.toFahrenheit()
    }
}

let c = 30.0
c.toFahrenheit()

Double.converToFahrenheit(from: 30.0)
```

extension String {
    subscript(idx: Int) -> String? {
        guard (0..<count).contains(idx) else {
                return nil
        }

    let target = index(startIndex, offsetBy: idx)
        return String(self[target])
    }
